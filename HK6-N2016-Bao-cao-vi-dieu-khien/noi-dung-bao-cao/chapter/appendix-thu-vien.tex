\item \verb|LCD_Init();| -- Khởi tạo LCD.

\item \verb|LCD_PutCmd(unsigned int cX);| -- Gửi lệnh lên LCD.
\item \verb|LCD_PulseEnable();| -- Xung kích hoạt.
\item \verb|LCD_setData(unsigned int cX);| -- Đặt dữ liệu lên chân Data.
\end{itemize}
